1) Username must be 8 to 20 characters in length
2) Username can only contain alphanumeric characters, numbers, underscore (_) and dot (.)
3) Username cannot start with underscore and dot

Password must contain at least:
1) 8 characters
2) 1 number
3) 1 upper case letter
4) 1 lower case letter
5) 1 special character

6) Password must not contain any spaces


<script language="JavaScript">
    function add() {
        var nodet = document.createElement('tr');
        var node = document.createElement('td');
        var nod = document.createElement('td');
        var no = document.createElement('td');
        node.innerHTML = "<input type='checkbox' name='1'>";
        nod.innerHTML = "<input type='text' name='2'>";
        no.innerHTML = "<input type='file' name='3'>";
        document.getElementById('tabl').appendChild(nodet);
        nodet.appendChild(node).appendChild(nod).appendChild(no);

    }

    function removeAllRowsContainingCheckedCheckbox(table) {
        for (var rowi= table.rows.length; rowi-->0;) {
            var row= table.rows[rowi];
            var inputs= row.getElementsByTagName('input');
            for (var inputi= inputs.length; inputi-->0;) {
                var input= inputs[inputi];

                if (input.type==='checkbox' && input.checked) {
                    row.parentNode.removeChild(row);
                    break;
                }
            }
        }
    }

</script>
